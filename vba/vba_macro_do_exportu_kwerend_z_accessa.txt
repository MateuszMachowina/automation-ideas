Sub EksportujZapytaniaDoFolderuZNazwąPliku()
    Dim qdf As QueryDef
    Dim db As DAO.Database
    Dim folderPath As String
    Dim fileName As String
    Dim exportFile As Integer
    Dim accFileName As String
    Dim fso As Object

    ' Nazwa pliku Accessa bez ścieżki i rozszerzenia
    accFileName = Mid(CurrentDb.Name, InStrRev(CurrentDb.Name, "\") + 1)
    accFileName = Left(accFileName, InStrRev(accFileName, ".") - 1)

    ' Folder na Pulpicie
    folderPath = Environ("USERPROFILE") & "\Desktop\AccessExport\"

    ' Tworzenie folderu jeśli nie istnieje
    Set fso = CreateObject("Scripting.FileSystemObject")
    If Not fso.FolderExists(folderPath) Then
        fso.CreateFolder folderPath
    End If

    ' Nazwa pliku z zapytaniami
    fileName = folderPath & "zapytania_" & accFileName & ".txt"

    ' Eksport zapytań
    Set db = CurrentDb
    exportFile = FreeFile
    Open fileName For Output As #exportFile

    For Each qdf In db.QueryDefs
        If Left(qdf.Name, 1) <> "~" Then
            Print #exportFile, "-- " & qdf.Name
            Print #exportFile, qdf.SQL
            Print #exportFile, ""
        End If
    Next qdf

    Close #exportFile

    MsgBox "Zapisano zapytania jako: " & fileName, vbInformation
End Sub
